-- начало работы над новым проектом
git init                   -- связать папку проэкта с github
git remote add origin      -- связать наш проэкт с проэктом на github

-- внести изменения
git add 				   -- указать измененне файлы 
git commit -m "some text"  -- создать изменения и наиминовть его 
git push -u origin main    -- указываем в какую ветку внести изменения

-- начало работы над существующим проектом
git clone                  --скопировать сущ-ий проэкт с github
    -- перейти в папку с проектом
npm install                -- установка библиотек npm  (все зависимости из packege.json в проекте)
